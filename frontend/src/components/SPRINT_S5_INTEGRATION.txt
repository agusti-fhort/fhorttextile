     {showWizard && (
       <SizingProfileWizard
         onComplete={(result) => {
           setSizingResult(result)
           setShowWizard(false)
           // Actualitzar el form amb els valors:
           setForm(f => ({
             ...f,
             size_system: result.size_system_id,
             grading_rule_set: result.grading_rule_set_id,
             base_size_label: result.base_size_label,
             size_run_model: result.size_run_model,
           }))
         }}
         onCancel={() => setShowWizard(false)}
       />
     )}

     {sizingResult && !showWizard && (
       <div style={{ padding:"10px", border:"1px solid #e0d5c5", borderRadius:6 }}>
         <div>{sizingResult.size_system_nom} · {sizingResult.base_size_label} ★</div>
         <div style={{ fontSize:11, color:"#868685" }}>{sizingResult.size_run_model}</div>
         <button onClick={() => setShowWizard(true)}>Canviar</button>
